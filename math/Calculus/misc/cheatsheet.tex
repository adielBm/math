\section{Series (Convergence Tests)}
\begin{tabular}{@{}ll@{}l@{}}
Divergence Test & $\lim_{n\to\infty} a_n \neq 0 \implies \sum a_n$ diverges \\
Geometric Series & $\sum_{n=0}^{\infty} ar^n = \frac{a}{1-r}$ if $|r|<1$ \\
P-Series & $\sum_{n=1}^{\infty} \frac{1}{n^p}$ converges if $p>1$ \\
Integral Test & $\int_{1}^{\infty} f(x) \, dx$ and $\sum_{n=1}^{\infty} f(n)$ converge together \\
Cauchy C.T. & $\sum a_n$ conv. iff $\forall \varepsilon > 0, \exists N: \forall n \geq N, \forall p \in \mathbb{N}, \sum_{k=n+1}^{n+p} a_k < \varepsilon$ \\
Absolute C.T. & $\sum |a_n|$ converges $\implies \sum a_n$ converges \\
MCT \\

\end{tabular}

\subsection{Comparison Test}
\textbf{Direct Comparison} ($\forall n \geq N, 0 \leq a_n \leq b_n$) 
\begin{itemize}
\item $\sum b_n$ converges $\implies \sum a_n$ converges
\item $\sum a_n$ diverges $\implies \sum b_n$ diverges (both to infinity)
\end{itemize}

\textbf{Limit Comparison} 
($\displaystyle\lim_{n\to\infty} \frac{a_n}{b_n} = c > 0$)
\begin{itemize}
\item $\sum b_n$ converges $\iff \sum a_n$ converges
\item  $\sum b_n$ diverges $\iff \sum a_n$ diverges 
\end{itemize}

\subsubsection{Root Test (Cauchy)}

\begin{itemize}
\item (5.16) 
\begin{enumerate}
	\item (a.) If there exists $q<1$ such that $\sqrt[n]{|a_n|}\leq q$ for almost all $n$, then the series $\sum a_n$ converges absolutely. 
	\item (b.) If $\sqrt[n]{|a_n|}\geq 1$ for infinitely many $n$, then the series $\sum a_n$ diverges.
\end{enumerate}
\item (5.16*) Given $c=\displaystyle\limsup_{n\to\infty}\sqrt[n]{|a_n|}$. or ($c=\displaystyle\lim_{n\to\infty}\sqrt[n]{|a_n|}$ exists (5.16**))
\begin{itemize}
    \item If $c<1$, then the series $\sum a_n$ converges absolutely.
    \item If $c>1$, then the series $\sum a_n$ diverges.
    \item If $c=1$, then the test is inconclusive.
\end{itemize}
\end{itemize}

\subsubsection{Ratio Test (d'Alembert)}
Given $a_n\neq0$ for all $n$ \\

\begin{itemize}
\item (5.17)
\begin{itemize}
    \item (a.) If there exists $q<1$ such that $\left|\frac{a_{n+1}}{a_n}\right|\leq q$ for almost all $n$, then the series $\sum a_n$ converges absolutely.
    \item (b.) If $\left|\frac{a_{n+1}}{a_n}\right|\geq 1$ for almost all $n$, then the series $\sum a_n$ diverges.    
\end{itemize}
\item (5.17**) Given $c=\displaystyle\lim_{n\to\infty}\left|\frac{a_{n+1}}{a_n}\right|$.
\begin{itemize}
    \item If $c<1$, then the series $\sum a_n$ converges absolutely.
    \item If $c>1$, then the series $\sum a_n$ diverges.
    \item If $c=1$, then the test is inconclusive.
\end{itemize}
\item (q5.26) If there exists $q<1$ such that $\left|\frac{a_{n+1}}{a_n}\right|\leq q<1$ for almost all $n$, then there exists $q\leq q'<1$ such that $\sqrt[n]{|a_n|}\leq q'$ for almost all $n$.
\end{itemize}



\subsubsection{Cauchy Condensation Test}

- (5.18) Cauchy condensation test - Let $(a_n)$ be a decreasing sequence of nonnegative terms. The series $\sum_{n=1}^{\infty}a_n$ converges if and only if the series $\sum_{n=1}^{\infty}2^na_{2^n}$ converges.

\subsubsection{Alternating Series Test (Leibniz)}

- (5.20) Let $(a_n)$ be a decreasing, null, (thus nonnegative) sequence. Then:
	- (A.) The series $\sum_{n=1}^{\infty}(-1)^{n+1}a_n$ converges. (This is a special case of [[#Dirichlet's test]]. see q5.32)
	- Alternating Series Estimation Theorem - If $S=\sum (-1)^{n+1}a_n$, and $S_n$ is the $n$-th partial sum, then for all $n$ we have:
		- (B.) $S$ is between $S_n$ and $S_{n+1}$.
		- (C.) The error $|S-S_n|$ is less than $a_{n+1}$.
